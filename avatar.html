How It Works

This JavaScript function extracts the first letter of the first and last names to create initials for the avatar.

If only one name is provided, it uses the first two letters of that name. Special characters and extra spaces are handled automatically.

// Function to generate initials from name
function generateInitials(firstName, lastName) {
    // Remove extra spaces and get first letters
    let firstInitial = firstName ? firstName.trim().charAt(0).toUpperCase() : '';
    let lastInitial = lastName ? lastName.trim().charAt(0).toUpperCase() : '';
    
    // If only one name provided, use first two letters
    if (firstInitial && !lastInitial && firstName.length > 1) {
        lastInitial = firstName.trim().charAt(1).toUpperCase();
    }
    
    return firstInitial + lastInitial;
}

// Function to update the avatar
function updateAvatar() {
    const firstName = document.getElementById('firstName').value;
    const lastName = document.getElementById('lastName').value;
    const avatar = document.getElementById('avatar');
    const profileName = document.getElementById('profileName');
    
    // Generate display name
    const displayName = (firstName || lastName) 
        ? `${firstName || ''} ${lastName || ''}`.trim() 
        : 'User Name';
    
    profileName.textContent = displayName;
    
    // Generate and set initials
    const initials = generateInitials(firstName, lastName);
    
    if (initials) {
        avatar.innerHTML = initials;
        avatar.style.fontSize = '3.5rem';
    } else {
        // Fallback to icon if no name provided
        avatar.innerHTML = '<i class="fas fa-user"></i>';
        avatar.style.fontSize = '4rem';
    }
}

// Event listeners
document.getElementById('updateAvatar').addEventListener('click', updateAvatar);

// Add event listeners to example avatars
document.querySelectorAll('.example-avatar').forEach(avatar => {
    avatar.addEventListener('click', function() {
        document.getElementById('firstName').value = this.dataset.first;
        document.getElementById('lastName').value = this.dataset.last;
        updateAvatar();
    });
});
